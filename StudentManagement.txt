	public static void removeStudent(Student[] arrayStudent , int SISO, String id)
	{
		for( int i = 0 ; i < SISO ; i ++ )
		{
			int sinhvienthuN;
			if (arrayStudent[i].getID() == id )
			{
				sinhvienthuN = i;	
				for(int j = i ; j < SISO ; j++)
				{
					arrayStudent[j] = arrayStudent[j+1];
				}
				SISO = SISO - 1;
				break;
			}
		}
	}
